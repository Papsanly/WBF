// Завдання 1: Перетворення Масиву
const inputArray = [10, 12, 15, 6, 9, 18, 3, 7, 14]
document.getElementById('originalArray').textContent = inputArray.join(', ')

function rearrangeArray(arr) {
  const divisibleBy2 = arr.filter(num => num % 2 === 0 && num % 3 !== 0)
  const divisibleBy2And3 = arr.filter(num => num % 2 === 0 && num % 3 === 0)
  const divisibleBy3 = arr.filter(num => num % 2 !== 0 && num % 3 === 0)
  return [...divisibleBy2, ...divisibleBy2And3, ...divisibleBy3]
}

const rearrangedArray = rearrangeArray(inputArray)
document.getElementById('rearrangedArray').textContent = rearrangedArray.join(', ')

// Завдання 2: Сортування Масиву (Швидке Сортування)
const inputArraySort = [...inputArray]
document.getElementById('originalArraySort').textContent = inputArraySort.join(', ')

function quickSort(arr) {
  if (arr.length <= 1) {
    return arr
  }

  const pivot = arr[Math.floor(arr.length / 2)]
  const less = arr.filter(num => num < pivot)
  const equal = arr.filter(num => num === pivot)
  const greater = arr.filter(num => num > pivot)

  return [...quickSort(less), ...equal, ...quickSort(greater)]
}

const sortedArray = quickSort(inputArraySort)
document.getElementById('sortedArray').textContent = sortedArray.join(', ')
