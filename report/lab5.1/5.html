class Client {
  #login;
  #email;

  constructor(login, email) {
    this.#login = login;
    this.#email = email;
  }

  // Getter for login
  get login() {
    return this.#login;
  }

  // Setter for login
  set login(newLogin) {
    this.#login = newLogin;
  }

  // Getter for email
  get email() {
    return this.#email;
  }

  // Setter for email
  set email(newEmail) {
    this.#email = newEmail;
  }
}

// Function to test the Client class
function testClient() {
  const client = new Client('user123', 'user123@example.com');
  console.log('Initial login:', client.login);
  console.log('Initial email:', client.email);

  client.login = 'newUser456';
  client.email = 'newUser456@example.com';
  console.log('Updated login:', client.login);
  console.log('Updated email:', client.email);
}
