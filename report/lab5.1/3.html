function logResults() {
  const medicines = {
    'Агалгін': new Date('2022-05-01'),
    'Ношпа': new Date('2025-07-02'),
    'Альфахолін': new Date('2024-12-21'),
    'Аспірин': new Date('2022-08-15'),
    'Аспаркам': new Date('2024-04-18'),
  }

  // Get the current date
  const currentDate = new Date()

  // Extract an array of medicine names and expiration dates
  const medicineArray = Object.entries(medicines)
    .filter(([name, expirationDate]) => expirationDate > currentDate) // Filter out expired medicines
    .sort((a, b) => a[1] - b[1]) // Sort medicines by expiration date
    .map(([name, expirationDate]) => name) // Extract just the names

  // Log the resulting array
  console.log(medicineArray)
}
