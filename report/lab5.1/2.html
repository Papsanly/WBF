// Function to create a product with a unique id and call the callback with the product
function createProduct(obj, callback) {
  // Adding a unique id to the product object
  obj.id = crypto.randomUUID()
  // Calling the callback function with the product object
  callback(obj)
}

// Callback to log the product
function logProduct(product) {
  console.log('Product created:', product)
}

// Callback to log the total price of the product
function logTotalPrice(product) {
  if (product.price && product.quantity) {
    const totalPrice = product.price * product.quantity
    console.log('Total price:', totalPrice)
  } else {
    console.log('Price or quantity not defined.')
  }
}

// Example product object
const product = {
  name: 'Sample Product',
  price: 10,
  quantity: 3
}

// Adding event listener to the button
document.getElementById('createProductButton').addEventListener('click', function () {
  createProduct(product, function (createdProduct) {
    logProduct(createdProduct)
    logTotalPrice(createdProduct)
  })
})
